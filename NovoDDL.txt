CREATE DATABASE newhotel;

USE newhotel;

CREATE TABLE RECEPCIONISTAS (
    CODIGO INT PRIMARY KEY AUTO_INCREMENT
    ,NOME VARCHAR (255) NOT NULL
    ,CPF CHAR(11) NOT NULL
    ,RG VARCHAR (10) NOT NULL
    ,SEXO CHAR(1) NOT NULL
    ,DATANASC DATETIME NOT NULL
    ,EMAIL VARCHAR (255) NOT NULL
    ,TEL VARCHAR (50)
    ,CEL VARCHAR (50)
    ,ENDERECO VARCHAR (255) NOT NULL
    ,NUM INT NOT NULL
    ,CEP CHAR(8) NOT NULL
    ,CIDADE VARCHAR (255) NOT NULL
    ,UF CHAR(2) NOT NULL
    ,SENHA VARCHAR (255) NOT NULL
);

CREATE TABLE QUARTOS (
    CODIGO INT PRIMARY KEY AUTO_INCREMENT
    ,DESCRICAO VARCHAR (255) NOT NULL
    ,VALOR INT NOT NULL
    ,QUANTPESSOAS INT NOT NULL
    ,QUANTQUARTOS INT NOT NULL
);

CREATE TABLE HOSPEDES (
    CODIGO INT PRIMARY KEY AUTO_INCREMENT
    ,NOME VARCHAR (255) NOT NULL
    ,CPF CHAR(11) NOT NULL
    ,RG VARCHAR (10) NOT NULL
    ,SEXO CHAR(1) NOT NULL
    ,DATANASC DATETIME NOT NULL
    ,EMAIL VARCHAR (255) NOT NULL
    ,TEL VARCHAR (50)
    ,CEL VARCHAR (50)
    ,ENDERECO VARCHAR (255) NOT NULL
    ,NUM INT NOT NULL
    ,CEP CHAR(8) NOT NULL
    ,CIDADE VARCHAR (255) NOT NULL
    ,UF CHAR(2) NOT NULL
);

CREATE TABLE RESERVASS (
    CODIGO INT PRIMARY KEY AUTO_INCREMENT
    ,COD_HOSPEDE INT NOT NULL
    ,DATAEN DATETIME NOT NULL
    ,DATASA DATETIME NOT NULL
    ,TIPOQUARTO VARCHAR (255) NOT NULL
    ,NPESSOAS INT NOT NULL
    ,VALOR VARCHAR (10) NOT NULL
    ,PAGAMENTO VARCHAR (100) NOT NULL
    ,FOREIGN KEY(COD_HOSPEDE) REFERENCES HOSPEDES(CODIGO)
);
